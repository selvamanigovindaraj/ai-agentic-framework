 42        if not self.id:
 43            self.id = f"agent_{uuid.uuid4().hex[:8]}"

 48    async def execute(self, task: str, context: Optional[Dict] = None) -> Dict[str, Any]:
 49        """Execute task using ReAct reasoning loop"""
 50        self.iteration_count = 0
 51        self.state.update({"task": task, "status": "running"})
 52        
 53        try:
 54            # Memory retrieval
 55            if self.memory:
 56                memories = await self.memory.retrieve(task, k=5)
 57                context = context or {}
 58                context["memories"] = memories
 59            
 60            # Build messages
 61            messages = self._build_messages(task, context)
 62            
 63            # Execute ReAct loop
 64            result = await self._react_loop(messages)
 65            
 66            # Store experience
 67            if self.memory:
 68                await self.memory.store({
 69                    "task": task,
 70                    "content": result,
 71                    "timestamp": datetime.now().isoformat()
 72                })
 73            
 74            self.state.update({"status": "completed"})
 75            return {
 76                "success": True,
 77                "output": result,
 78                "iterations": self.iteration_count,
 79                "cost": self.total_cost
 80            }
 81            
 82        except Exception as e:
 83            self.state.update({"status": "failed", "error": str(e)})
 84            return {
 85                "success": False,
 86                "error": str(e),
 87                "iterations": self.iteration_count
 88            }

 90    def _build_messages(self, task: str, context: Optional[Dict] = None) -> List[Dict]:
 91        """Build LLM messages with tools and context"""
 92        messages = [{"role": "system", "content": self.instructions}]
 93        
 94        if context and context.get("memories"):
 95            memories_text = "\n".join([m["content"] for m in context["memories"]])
 96            messages.append({"role": "system", "content": f"Relevant memories:\n{memories_text}"})
 97        
 98        if self.tools:
 99            tools_text = self._format_tools()
100            messages.append({"role": "system", "content": f"Available tools:\n{tools_text}"})
101        
102        messages.append({"role": "user", "content": task})
103        return messages

105    def _format_tools(self) -> str:
106        """Format tools for prompt"""
107        return "\n".join([f"- {t.name}: {t.description}" for t in self.tools])

109    async def _react_loop(self, messages: List[Dict]) -> str:
110        """ReAct: Thought  Action  Observation loop"""
111        conversation = messages.copy()
112        
113        while self.iteration_count < self.config.max_iterations:
114            self.iteration_count += 1
115            
116            response = await self.model_client.generate(
117                messages=conversation,
118                temperature=self.config.temperature
119            )
120            
121            self.total_cost += response.get("cost", 0)
122            
123            if self.total_cost > self.config.max_cost:
124                raise Exception(f"Cost limit exceeded: ${self.total_cost:.2f}")
125            
126            content = response["content"]
127            conversation.append({"role": "assistant", "content": content})
128            
129            if self._is_task_complete(content):
130                return content
131            
132            if tool_calls := self._extract_tool_calls(content):
133                for tool_call in tool_calls:
134                    observation = await self._execute_tool(tool_call)
135                    conversation.append({
136                        "role": "user",
137                        "content": f"Observation: {observation}"
138                    })
139            else:
140                return content
141        
142        raise Exception(f"Max iterations ({self.config.max_iterations}) exceeded")

144    def _is_task_complete(self, response: str) -> bool:
145        """Check for task completion markers"""
146        markers = ["final answer", "task complete", "conclusion", "finished"]
147        return any(marker in response.lower() for marker in markers)

149    def _extract_tool_calls(self, response: str) -> List[Dict]:
150        """Extract tool calls using AST parsing"""
151        import re
152        import ast
153        import inspect
154        
155        tool_calls = []
156        for tool in self.tools:
157            # Match tool_name(...) handling multiline args
158            # We use a simplified regex to find the start, then let AST handle parsing
159            # This regex finds "tool_name(" and captures until ")"
160            # Note: This is still a bit fragile for nested parens, but better than before
161            pattern = rf"{tool.name}\s*\((?:[^)(]|\((?:[^)(]+|\([^)(]*\))*\))*\)"
162            
163            for match in re.finditer(pattern, response, re.IGNORECASE):
164                call_str = match.group(0)
165                try:
166                    # Parse the call string
167                    tree = ast.parse(call_str, mode='eval')
168                    if not isinstance(tree.body, ast.Call):
169                        continue
170                        
171                    # Extract arguments
172                    params = {}
173                    
174                    # Handle keyword args (key=value)
175                    for kw in tree.body.keywords:
176                        params[kw.arg] = ast.literal_eval(kw.value)
177                    
178                    # Handle positional args
179                    if tree.body.args:
180                        # Inspect tool signature to map positional args to names
181                        sig = inspect.signature(tool.execute)
182                        param_names = list(sig.parameters.keys())
183                        
184                        for i, arg in enumerate(tree.body.args):
185                            if i < len(param_names):
186                                params[param_names[i]] = ast.literal_eval(arg)
187                    
188                    tool_calls.append({"tool": tool, "params": params})
189                    
190                except Exception:
191                    # Fallback or ignore invalid calls
192                    continue
193                    
194        return tool_calls
195    
196    async def _execute_tool(self, tool_call: Dict) -> str:
197        """Execute tool and return observation"""
198        tool = tool_call["tool"]
199        try:
200            result = await tool.execute(**tool_call["params"])
201            return f"Tool {tool.name} returned: {result}"
202        except Exception as e:
203            return f"Tool {tool.name} error: {str(e)}"

206        """Get current agent state"""

14    def register(self, tool: Tool) -> None:
15        """Register a tool"""
16        self.tools[tool.id] = tool
17        
18        if tool.category not in self.categories:
19            self.categories[tool.category] = []
20        self.categories[tool.category].append(tool.id)
21        
22        print(f" Registered tool: {tool.name} (ID: {tool.id})")

24    def get(self, tool_id: str) -> Optional[Tool]:
25        """Get tool by ID"""
26        return self.tools.get(tool_id)
27    
28    def get_all(self) -> List[Tool]:
29        """Get all registered tools"""
30        return list(self.tools.values())
31    
32    def discover(self, query: str, filters: Optional[Dict] = None) -> List[Tool]:
33        """Discover tools by query and filters"""
34        results = []
35        
36        for tool in self.tools.values():
37            if (query.lower() in tool.name.lower() or 
38                query.lower() in tool.description.lower()):
39                
40                if filters:
41                    if "category" in filters and tool.category != filters["category"]:
42                        continue
43                
44                results.append(tool)
45        
46        return results
47    
48    def list_categories(self) -> List[str]:
49        """List all tool categories"""
50        return list(self.categories.keys())
51    
52    def get_by_category(self, category: str) -> List[Tool]:
53        """Get tools by category"""
54        tool_ids = self.categories.get(category, [])
55        return [self.tools[tid] for tid in tool_ids if tid in self.tools]
